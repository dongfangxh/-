/*1.	查询学生表中的数据。*/
USE 学生管理
SELECT *
FROM 学生表
/*2.	查询学生表中的学号、姓名、入学成绩。*/
SELECT 学号,姓名,入学成绩
FROM 学生表
/*3.	查询学生表中所有男生的信息。*/
SELECT *
FROM 学生表
WHERE 性别='男'
/*4.	给学生表中所有学生的入学成绩加10分。*/
UPDATE 学生表
SET 入学成绩=入学成绩+10
/*5.	查询学生表中1999年出生的学生姓名和出生日期。*/
SELECT 姓名,出生日期
FROM 学生表
WHERE YEAR(出生日期)='1999'
/*6.	查询学生表中所有姓李的学生信息。*/
SELECT *
FROM 学生表
WHERE 姓名 LIKE '李%'
/*7.	查询学生表中的学号、姓名、年龄，并给学号列别名学生编号。*/
SELECT 学号 AS 学生编号,姓名,YEAR(GETDATE())-YEAR(出生日期) AS 年龄
FROM 学生表
/*8.	查询学生表中前3条记录。*/
SELECT TOP 3 *
FROM 学生表
/*9.	查询学生表中性别列的值，去掉重复值。*/
SELECT DISTINCT 性别
FROM 学生表
/*10.	查询入学成绩在410到480分之间的学生姓名和入学成绩。*/
SELECT 姓名,入学成绩
FROM 学生表
WHERE 入学成绩 BETWEEN 410 AND 480
/*11.	查询入学成绩不在460分以上学生的信息。*/
SELECT 姓名,入学成绩
FROM 学生表
WHERE 入学成绩 <=460
/*12.	查询姓张、王、李三大姓的学生信息。*/
SELECT *
FROM 学生表
WHERE 姓名 LIKE '张%' OR 姓名 LIKE '王%' OR 姓名 LIKE '李%' 
/*13.	查询入学成绩在500分以下的男生信息。*/
SELECT *
FROM 学生表
WHERE 入学成绩<500 AND 性别='男'
/*14.	查询入学成绩在460分以下和所有女生信息，并按学号降序排序。*/
SELECT *
FROM 学生表
WHERE 入学成绩<460 OR 性别='女'
ORDER BY 学号 DESC
